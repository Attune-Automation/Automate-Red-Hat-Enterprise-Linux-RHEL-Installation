if [ ! -d "{rpmCaCertificateDropInDirectory}" ]
then
    echo "{rpmCaCertificateDropInDirectory} does not exist, skipping."
else
    cd "{rpmCaCertificateDropInDirectory}"
    
    NUMBER_FILES=$(ls -1q | wc -l)
    echo "NUMBER_FILES=${NUMBER_FILES}"
    
    if [ "${NUMBER_FILES}" -eq "0" ]
    then
        echo "No files in {rpmCaCertificateDropInDirectory}, skipping."
    else
        echo "{rpmCaCertificateDropInDirectory} exists, copying it's contents to {ksAttuneBaseDir}/build-{kickstartedNode.fqn}/cacerts."
        cp --verbose --recursive ./* "{ksAttuneBaseDir}/build-{kickstartedNode.fqn}/cacerts"
    fi
fi
