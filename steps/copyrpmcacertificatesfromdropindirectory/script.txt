DIR="{newOsRpmCaCertificateDirectory}"
if [ ! -d "$DIR" ]
then
    echo "$DIR does not exist, skipping."
else
    cd "$DIR"
    
    NUMBER_FILES=$(ls -1q | wc -l)
    echo "NUMBER_FILES=${NUMBER_FILES}"
    
    if [ "${NUMBER_FILES}" -eq "0" ]
    then
        echo "No files in $DIR, skipping."
    else
        echo "$DIR exists, copying it's contents to {automationWorkerBaseDir}/build-{newOsLinuxNode.fqn}/cacerts."
        cp --verbose --recursive ./* "{automationWorkerBaseDir}/build-{newOsLinuxNode.fqn}/cacerts"
    fi
fi
